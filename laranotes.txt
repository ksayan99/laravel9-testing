create laravel artisan command - composer global require laravel/installer

create migration - php artisan make:migration create_customname_table

file specific migration - php artisan migrate --path=/database/migrations/filename.php

run migration with seeder - php artisan migrate:fresh --seed

file specific seeder command - php artisan db:seed --class=Filename

create controller with resource - php artisan make:controller controller_name -r

change visibility of error files - php artisan vendor:publish --tag==laravel-errors

whitelist page maintanace mode - php artisan down --secret='keyword'
